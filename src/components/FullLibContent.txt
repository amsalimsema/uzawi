'use client';

import { motion } from 'framer-motion';
import { FiEye, FiDownload, FiX } from 'react-icons/fi';
import { useState, useEffect } from 'react';
import { createClient } from 'contentful';

export default function ResearchLibrary() {
  // Contentful client
  const client = createClient({
    space: import.meta.env.VITE_CONTENTFUL_SPACE_ID,
    accessToken: import.meta.env.VITE_CONTENTFUL_ACCESS_KEY,
  });

  // State for data loaded from Contentful
  const [researchItems, setResearchItems] = useState([]);

  // Load data from Contentful
  useEffect(() => {
    async function loadData() {
      try {
        const res = await client.getEntries({
          content_type: 'researchItem',
          order: '-sys.createdAt',
        });

        const mapped = res.items.map((item) => ({
          id: item.sys.id,
          title: item.fields.title,
          description: item.fields.description,
          category: item.fields.category,
          pdfUrl: item.fields.pdfUrl,
          image: item.fields.image?.fields?.file?.url
            ? `https:${item.fields.image.fields.file.url}`
            : '/assets/fallback.jpg', // Fallback image from assets
          borderColor: item.fields.borderColor || '#00CAFF',
        }));

        setResearchItems(mapped);
      } catch (error) {
        console.error('Contentful fetch error:', error);
      }
    }

    loadData();
  }, []);

  // PDF modal
  const [previewPdf, setPreviewPdf] = useState(null);

  // Category filter state
  const [activeCategory, setActiveCategory] = useState('All');

  const categories = [
    'All',
    'Surveillance',
    'Information Integrity',
    'AI & Resources',
    'Digital Rights',
  ];

  const filteredResearch =
    activeCategory === 'All'
      ? researchItems
      : researchItems.filter((item) => item.category === activeCategory);

  const handlePreview = (pdfUrl) => {
    setPreviewPdf(pdfUrl);
    document.body.style.overflow = 'hidden';
  };

  const handleClosePreview = () => {
    setPreviewPdf(null);
    document.body.style.overflow = 'unset';
  };

  const handleDownload = (pdfUrl, title) => {
    const link = document.createElement('a');
    link.href = pdfUrl;
    link.download = `${title.replace(/\s+/g, '-').toLowerCase()}.pdf`;
    document.body.appendChild(link);
    link.click();
    document.body.removeChild(link);
  };

  const containerVariants = {
    hidden: { opacity: 0 },
    visible: {
      opacity: 1,
      transition: {
        staggerChildren: 0.1,
        delayChildren: 0.1,
      },
    },
  };

  const itemVariants = {
    hidden: { opacity: 0, y: 20 },
    visible: {
      opacity: 1,
      y: 0,
      transition: { duration: 0.5, ease: 'easeOut' },
    },
  };

  return (
    <section className="relative bg-white overflow-hidden">
      {/* --- HEADER SECTION (unchanged) --- */}
      <div className="relative bg-gradient-to-br from-[#0F1662] via-[#1a2580] to-[#4065FF] overflow-hidden">
        <div className="absolute inset-0 opacity-20">
          <div
            className="absolute inset-0"
            style={{
              backgroundImage: `linear-gradient(#00CAFF 1px, transparent 1px), linear-gradient(90deg, #00CAFF 1px, transparent 1px)`,
              backgroundSize: '40px 40px',
            }}
          />
        </div>

        <div className="absolute inset-0 bg-gradient-to-b from-transparent via-[#0F1662]/50 to-[#0F1662]" />
        <div className="absolute inset-0 bg-gradient-to-r from-[#0F1662]/30 to-transparent" />

        <div className="absolute top-0 left-0 right-0 h-16 flex items-center justify-around overflow-hidden">
          {[...Array(20)].map((_, i) => (
            <div key={i} className="flex items-center gap-2">
              <div className="w-8 h-8 border-2 border-white/30 rotate-45" />
              <div className="w-3 h-3 rounded-full bg-[#00CAFF]/30" />
            </div>
          ))}
        </div>

        <div className="relative max-w-7xl mx-auto px-4 sm:px-6 lg:px-8 pt-24 pb-16">
          <motion.div
            initial={{ opacity: 0, y: 20 }}
            animate={{ opacity: 1, y: 0 }}
            transition={{ duration: 0.8 }}
            className="text-center"
          >
            <h1 className="text-4xl md:text-5xl lg:text-6xl mb-4">
              <span
                className="text-transparent"
                style={{
                  WebkitTextStroke: '2px #00CAFF',
                  color: 'transparent',
                }}
              >
                RESEARCH
              </span>
              <br />
              <span className="text-[#00CAFF]">LIBRARY</span>
            </h1>
            <p className="text-white/80 text-sm max-w-2xl mx-auto uppercase tracking-wide">
              Explore our comprehensive collection of research on AI, democracy,
              and digital sovereignty in Africa
            </p>
          </motion.div>
        </div>

        <div className="absolute bottom-0 left-0 right-0 h-16 flex items-center justify-around overflow-hidden">
          {[...Array(20)].map((_, i) => (
            <div key={i} className="flex items-center gap-2">
              <div className="w-6 h-6 bg-[#4065FF]/30" />
              <div className="w-3 h-3 rounded-full bg-white/30" />
              <div className="w-4 h-8 bg-[#00CAFF]/30 rotate-12" />
            </div>
          ))}
        </div>
      </div>

      {/* --- CATEGORY FILTER SECTION (unchanged) --- */}
      <div className="bg-gray-50 border-b border-gray-200 py-6 rounded-bl-3xl rounded-br-3xl">
        <div className="max-w-6xl mx-auto px-4 sm:px-6 lg:px-8">
          <div className="flex justify-center md:justify-center gap-3 overflow-x-auto scrollbar-hide pb-2 px-4">
            {categories.map((category) => (
              <button
                key={category}
                onClick={() => setActiveCategory(category)}
                className={`px-4 py-2 text-sm transition-all duration-200 rounded-sm whitespace-nowrap flex-shrink-0 ${
                  activeCategory === category
                    ? 'bg-[#4065FF] text-white'
                    : 'bg-white text-gray-700 hover:bg-gray-100 border border-gray-300'
                }`}
              >
                {category}
              </button>
            ))}
          </div>
        </div>
      </div>

      {/* --- RESEARCH GRID SECTION (unchanged except image fallback) --- */}
      <div className="relative py-16">
        <div
          className="absolute inset-0 opacity-[0.03]"
          style={{
            backgroundImage: `linear-gradient(#00CAFF 1px, transparent 1px), linear-gradient(90deg, #00CAFF 1px, transparent 1px)`,
            backgroundSize: '60px 60px',
          }}
        />

        <motion.div
          className="relative max-w-6xl mx-auto px-4 sm:px-6 lg:px-8"
          variants={containerVariants}
          initial="hidden"
          animate="visible"
          key={activeCategory}
        >
          <div className="grid md:grid-cols-2 gap-8">
            {filteredResearch.map((item) => (
              <motion.div
                key={item.id}
                className="relative bg-white p-5 border hover:shadow-lg transition-all duration-300 text-center md:text-left rounded-r-3xl"
                style={{
                  borderLeftColor: item.borderColor,
                  borderLeftWidth: '1px',
                }}
                variants={itemVariants}
                whileHover={{ x: 4 }}
              >
                <div className="w-full h-40 mb-4 overflow-hidden flex items-center justify-center md:justify-start">
                  <img
                    src={item.image || '/assets/fallback.jpg'}
                    alt={item.title}
                    className="w-full h-full object-cover rounded-lg"
                  />
                </div>

                <h3
                  className="text-xl mb-2 font-serif"
                  style={{ color: '#0F1662' }}
                >
                  {item.title}
                </h3>

                <p className="text-gray-600 text-sm leading-relaxed mb-4">
                  {item.description}
                </p>

                <div className="flex items-center justify-center md:justify-start gap-6 mt-4">
                  <button
                    onClick={() => handlePreview(item.pdfUrl)}
                    className="flex items-center gap-2 text-gray-600 hover:text-[#4065FF] transition-colors duration-200 p-2 -m-2"
                  >
                    <FiEye size={22} className="md:w-5 md:h-5" />
                    <span className="text-sm md:text-base">Preview</span>
                  </button>

                  <button
                    onClick={() => handleDownload(item.pdfUrl, item.title)}
                    className="flex items-center gap-2 text-gray-600 hover:text-[#00CAFF] transition-colors duration-200 p-2 -m-2"
                  >
                    <FiDownload size={22} className="md:w-5 md:h-5" />
                    <span className="text-sm md:text-base">Download</span>
                  </button>
                </div>
              </motion.div>
            ))}
          </div>

          {filteredResearch.length === 0 && (
            <motion.div
              className="text-center py-16"
              initial={{ opacity: 0 }}
              animate={{ opacity: 1 }}
            >
              <p className="text-gray-500 text-lg">
                No research found in this category.
              </p>
            </motion.div>
          )}
        </motion.div>
      </div>

      {/* --- PDF PREVIEW MODAL (unchanged) --- */}
      {previewPdf && (
        <div className="fixed inset-0 z-50 flex items-center justify-center bg-black/90 p-0 sm:p-4">
          <button
            onClick={handleClosePreview}
            className="absolute top-2 right-2 sm:top-4 sm:right-4 z-10 text-white hover:text-gray-300 transition-colors p-2 bg-black/50 rounded-full"
          >
            <FiX size={28} className="sm:w-8 sm:h-8" />
          </button>

          <div className="w-full h-full sm:max-w-6xl sm:max-h-[90vh] bg-white sm:rounded-lg overflow-hidden">
            <iframe
              src={previewPdf}
              className="w-full h-full"
              title="PDF Preview"
              allowFullScreen
            />
          </div>
        </div>
      )}
    </section>
  );
}

export { ResearchLibrary };

import { useState, useEffect } from "react";
import { createClient } from "contentful";

export default function FeaturedLib() {
  const [researchItems, setResearchItems] = useState([]);
  const [previewPdf, setPreviewPdf] = useState(null);

  // Fallback image from your /assets folder
  const fallbackImage = "/assets/fallback.jpg";

  // Contentful client
  const client = createClient({
    space: import.meta.env.VITE_CONTENTFUL_SPACE_ID,
    accessToken: import.meta.env.VITE_CONTENTFUL_ACCESS_TOKEN,
  });

  // Fetch featured items only
  useEffect(() => {
    const fetchFeatured = async () => {
      try {
        const response = await client.getEntries({
          content_type: "researchItem",
          "fields.featured": true,
          limit: 4,
        });

        const formatted = response.items.map((item) => ({
          id: item.sys.id,
          title: item.fields.title,
          description: item.fields.description,
          borderColor: item.fields.borderColor || "#00CAFF",
          image: item.fields.image?.fields.file.url
            ? `https:${item.fields.image.fields.file.url}`
            : fallbackImage,
          pdfUrl: item.fields.pdfFile?.fields.file.url
            ? `https:${item.fields.pdfFile.fields.file.url}`
            : null,
        }));

        setResearchItems(formatted);
      } catch (error) {
        console.error("Error fetching featured items:", error);
      }
    };

    fetchFeatured();
  }, []);

  const handlePreview = (pdfUrl) => {
    setPreviewPdf(pdfUrl);
    document.body.style.overflow = "hidden";
  };

  const handleClosePreview = () => {
    setPreviewPdf(null);
    document.body.style.overflow = "unset";
  };

  const handleDownload = (pdfUrl, title) => {
    const link = document.createElement("a");
    link.href = pdfUrl;

    visible: {
      opacity: 1,
      transition: { staggerChildren: 0.1, delayChildren: 0.1 },
    },
  };

  const itemVariants = {
    hidden: { opacity: 0, y: 20 },
    visible: { opacity: 1, y: 0, transition: { duration: 0.5 } },
  };

  return (
    <section className="relative bg-white py-10 overflow-hidden">
      <div
        className="absolute inset-0 opacity-[0.03]"
        style={{
          backgroundImage: `
            linear-gradient(to right, #00CAFF 1px, transparent 1px),
            linear-gradient(to bottom, #00CAFF 1px, transparent 1px)
          `,
          backgroundSize: "60px 60px",
        }}
      />

      <motion.div
        className="relative max-w-6xl mx-auto px-4 sm:px-6 lg:px-8"
        variants={containerVariants}
        initial="hidden"
        whileInView="visible"
        viewport={{ once: true, margin: "-50px" }}
      >
        <motion.div variants={itemVariants} className="mb-24">
          <div className="text-center mb-12">
            <h2
              className="text-3xl md:text-4xl font-serif mb-3"
              style={{ color: "#0F1662" }}
            >
              Featured Research Papers
            </h2>
            <p className="text-base text-gray-500 max-w-2xl mx-auto">
              Critical investigations into AI, democracy, and digital sovereignty
              in Africa
            </p>
          </div>

          <div className="grid md:grid-cols-2 gap-8 max-w-5xl mx-auto">
            {researchItems.map((item) => (
              <motion.div
                key={item.id}
                className="relative bg-white p-5 border hover:shadow-lg transition-all duration-300 text-center md:text-left rounded-r-3xl"
                style={{ borderLeftColor: item.borderColor, borderLeftWidth: 4 }}
                variants={itemVariants}

                    className="flex items-center gap-2 text-gray-600 hover:text-[#4065FF] transition"
                  >
                    <FiEye size={22} />
                    <span>Preview</span>
                  </button>

                  <button

                    className="flex items-center gap-2 text-gray-600 hover:text-[#00CAFF] transition"

        >
          <h3 className="text-2xl md:text-3xl font-serif text-white mb-4">

          </div>
        </div>
      )}
    </section>
  );
